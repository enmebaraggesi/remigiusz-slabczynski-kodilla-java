SELECT I.SUMMARY, J.NAME
FROM ISSUES I JOIN ISSUESLIST J ON I.ISSUESLIST_ID = J.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID
HAVING COUNT(*) >= 1
ORDER BY COUNT(*);
